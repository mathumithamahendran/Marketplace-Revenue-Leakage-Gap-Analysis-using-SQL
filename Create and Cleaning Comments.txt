Create table discounts (Discount_Id int,Order_Id int,Discount_Amount int,Discount_Type text,
Coupon_Code text,Is_Stackable text);

Create table logistics_cost(Logistics_Id int,Order_Id int,Shipping_Cost int,
Reverse_Shipping_Cost int,Delivery_Days int,Delivery_Status text);

Create table orders (Order_Id int,Order_Date date,Customer_Id int,Product_Id int,Quantity int,
Selling_Price int,Order_Status text,Payment_Method text,Order_Channel text,Warehouse_Id text);

Create table payment_fees(Payment_Method text,Fee_Percentage float,Settlement_Days int);

Create table products(Product_Id int,Category text,Sub_Category text,Brand text,Supplier_Id text,
Cost_Price int,Mrp int,Weight_Kg float,Launch_Date date);

Create table returns(Return_Id int,Order_Id int,Return_Flag text,Return_Reason text,
Return_Initiated_Date date,Refund_Mode text,Refund_Status text,Customer_Fault_Flag text);
